const API_BASE_URL = "https://gateway-service-production-2ae6.up.railway.app"; const token = localStorage.getItem('jwtToken'); const email = localStorage.getItem('userEmail'); const headers = { 'Content-Type': 'application/json', 'Authorization': `Bearer ${token}` }; // Función para evitar mostrar "null" o "undefined" const validar = (dato, reemplazo = 'PENDIENTE') => { return (dato && dato !== "null" && dato !== "undefined" && dato.toString().trim() !== "") ? dato : `${reemplazo}`; }; async function cargarPerfil() { const grid = document.getElementById('perfil-grid'); try { const res = await fetch(`${API_BASE_URL}/api/auth/me?email=${email}`, { headers }); if (res.ok) { const u = await res.json(); // 1. Datos de Cabecera document.getElementById('p-full-name').innerText = `${u.nombre} ${u.apellido || ''}`; document.getElementById('p-foto').src = `https://ui-avatars.com/api/?name=${u.nombre}+${u.apellido || ''}&background=random`; // 2. Badge de Rol (Extraído de window.userRole definido en security.js) const rolLimpio = window.userRole ? window.userRole.replace('ROLE_', '') : 'ALUMNO'; document.getElementById('p-badge-rol').innerText = `RANGO: ${rolLimpio}`; // 3. Cuadrícula de Datos con validación grid.innerHTML = `

DOCUMENTO DE IDENTIDAD (DNI)

${validar(u.dni)}

CORREO ELECTRÓNICO

${u.email}

DIRECCIÓN DE RESIDENCIA

${validar(u.direccion, 'DIRECCIÓN NO REGISTRADA')}

ESTADO EN EL DOJO

● CUENTA ACTIVA

`; // 4. Pre-llenar inputs de edición document.getElementById('edit-tel').value = u.telefono || ''; document.getElementById('edit-dir').value = u.direccion || ''; } } catch (e) { grid.innerHTML = `
Error al conectar con Auth-Service en Railway.
`; } } async function actualizarDatos() { const tel = document.getElementById('edit-tel').value.trim(); const dir = document.getElementById('edit-dir').value.trim(); if (!tel || !dir) { alert("⚠️ Completa teléfono y dirección para validar tu ficha."); return; } try { const res = await fetch(`${API_BASE_URL}/api/auth/update-profile`, { method: 'PUT', headers, body: JSON.stringify({ email, telefono: tel, direccion: dir }) }); if (res.ok) { alert("✅ ¡Datos sincronizados con Neon!"); cargarPerfil(); } } catch (e) { alert("Error de red."); } } function cerrarSesion() { localStorage.clear(); window.location.href = 'login.html'; } // --- 🛡️ CONTROL DE PESTAÑAS SEGÚN RANGO --- document.addEventListener('DOMContentLoaded', () => { const rol = window.userRole ? window.userRole.toUpperCase() : 'ROLE_ALUMNO'; // Si el usuario es ADMIN o PROFESOR, inyectamos las herramientas de gestión if (rol.includes('ADMIN') || rol.includes('PROFESOR')) { const navExtras = document.getElementById('nav-admin-extras'); if (navExtras) { navExtras.innerHTML = ` Gestión Alumnos Usuarios y Roles Sedes `; } } cargarPerfil(); });
